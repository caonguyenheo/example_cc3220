161 /*
162  * PWMTimer port/pin defines for pin configuration.
163  *
164  * The timer id (0, 1, 2, or 3) is stored in bits 31 - 28
165  * The timer half (0 = A, 1 = B) is stored in bits 27 - 24
166  * The GPIO port (0, 1, 2, or 3) is stored in bits 23 - 20
167  * The GPIO pin index within the port (0 - 7) is stored in bits 19 - 16
168  * The pin mode is stored in bits 11 - 8
169  * The pin number is stored in bits 7 - 0
170  *
171  *
172  * 31 - 28 27 - 24 23 - 20 19 - 16 11 - 8 7 - 0
173  * -----------------------------------------------------------------------
174  * | Timer id | Timer half | GPIO port | GPIO pin index | pin mode | pin |
175  * -----------------------------------------------------------------------
176  *
177  * The CC32XX has fixed GPIO assignments and pin modes for a given pin.
178  * A PWM pin mode for a given pin has a fixed timer/timer-half.
179  */
180 #define PWMTimerCC32XX_T0A (0x00 << 24)
181 #define PWMTimerCC32XX_T0B (0x01 << 24)
182 #define PWMTimerCC32XX_T1A (0x10 << 24)
183 #define PWMTimerCC32XX_T1B (0x11 << 24)
184 #define PWMTimerCC32XX_T2A (0x20 << 24)
185 #define PWMTimerCC32XX_T2B (0x21 << 24)
186 #define PWMTimerCC32XX_T3A (0x30 << 24)
187 #define PWMTimerCC32XX_T3B (0x31 << 24)
188 
189 #define PWMTimerCC32XX_GPIO9 (0x11 << 16)
190 #define PWMTimerCC32XX_GPIO10 (0x12 << 16)
191 #define PWMTimerCC32XX_GPIO11 (0x13 << 16)
192 #define PWMTimerCC32XX_GPIO24 (0x30 << 16)
193 #define PWMTimerCC32XX_GPIO25 (0x31 << 16)
194 
195 #define PWMTimerCC32XX_GPIONONE (0xFF << 16)
